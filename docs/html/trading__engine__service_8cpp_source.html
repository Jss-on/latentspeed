222 void TradingEngineService::processEvent(const ccapi::Event& event, ccapi::Session* sessionPtr) {
223  if (event.getType() == ccapi::Event::Type::SUBSCRIPTION_STATUS) {
224  std::cout << "[TradingEngine] Subscription status: " << event.toPrettyString(2, 2) << std::endl;
225  } else if (event.getType() == ccapi::Event::Type::SUBSCRIPTION_DATA) {
226  // Process market data - unchanged for now, can be enhanced later
227  for (const auto& message : event.getMessageList()) {
228  MarketDataSnapshot snapshot;
229 
230  // Extract correlation ID to determine exchange/instrument
231  auto correlationIds = message.getCorrelationIdList();
232  if (!correlationIds.empty()) {
233  std::string corrId = correlationIds[0];
234  size_t pos = corrId.find(':');
235  if (pos != std::string::npos) {
236  snapshot.exchange = corrId.substr(0, pos);
237  snapshot.instrument = corrId.substr(pos + 1);
238  }
239  }
240 
241  // Extract market data
242  for (const auto& element : message.getElementList()) {
243  const auto& nameValueMap = element.getNameValueMap();
244 
245  auto bidPrice = nameValueMap.find("BID_PRICE");
246  auto bidSize = nameValueMap.find("BID_SIZE");
247  auto askPrice = nameValueMap.find("ASK_PRICE");
248  auto askSize = nameValueMap.find("ASK_SIZE");
249 
250  if (bidPrice != nameValueMap.end()) snapshot.bid_price = std::stod(bidPrice->second);
251  if (bidSize != nameValueMap.end()) snapshot.bid_size = std::stod(bidSize->second);
252  if (askPrice != nameValueMap.end()) snapshot.ask_price = std::stod(askPrice->second);
253  if (askSize != nameValueMap.end()) snapshot.ask_size = std::stod(askSize->second);
254  }
255 
256  snapshot.timestamp = std::to_string(std::chrono::duration_cast<std::chrono::milliseconds>(
257  std::chrono::system_clock::now().time_since_epoch()).count());
258 
259  // TODO: Broadcast market data via contract-compliant method
260  std::cout << "[TradingEngine] Market data: " << snapshot.exchange << ":" << snapshot.instrument
261  << " bid=" << snapshot.bid_price << " ask=" << snapshot.ask_price << std::endl;
262  }
263  } else if (event.getType() == ccapi::Event::Type::RESPONSE) {
264  // Handle CEX order responses
265  for (const auto& message : event.getMessageList()) {
266  auto correlationIds = message.getCorrelationIdList();
267  if (!correlationIds.empty()) {
268  std::string cl_id = correlationIds[0];
269 
270  // Create ExecutionReport based on response
271  ExecutionReport report;
272  report.cl_id = cl_id;
273  report.ts_ns = get_current_time_ns();
274 
275  // Check if order was accepted or rejected
276  if (message.getType() == ccapi::Message::Type::CREATE_ORDER) {
277  auto elements = message.getElementList();
278  if (!elements.empty()) {
279  const auto& element = elements[0];
280  const auto& nameValueMap = element.getNameValueMap();
281 
282  auto orderIdIt = nameValueMap.find("ORDER_ID");
283  if (orderIdIt != nameValueMap.end()) {
284  report.status = "accepted";
285  report.reason_code = "ok";
286  report.reason_text = "Order accepted by exchange";
287  report.exchange_order_id = orderIdIt->second;
288 
289  // Store mapping for future reference
290  cex_order_mapping_[cl_id] = orderIdIt->second;
291  } else {
292  report.status = "rejected";
293  report.reason_code = "venue_reject";
294  report.reason_text = "Order rejected by exchange";
295  }
296  }
297  } else {
298  report.status = "rejected";
299  report.reason_code = "venue_reject";
300  report.reason_text = "Order rejected by exchange";
301  }
302 
303  publish_execution_report(report);
304  }
305  }
306  }

389  }

410 void TradingEngineService::process_execution_order(const ExecutionOrder& order) {
411  try {
412  // Check for duplicate orders (idempotency)
413  if (is_duplicate_order(order.cl_id)) {
414  std::cout << "[TradingEngine] Ignoring duplicate order: " << order.cl_id << std::endl;
415  return;
416  }
417 
418  // Mark order as processed
419  mark_order_processed(order.cl_id);
420 
421  // Store pending order for tracking
422  pending_orders_[order.cl_id] = order;
423 
424  std::cout << "[TradingEngine] Processing " << order.action << " order: "
425  << order.venue_type << "/" << order.product_type << " cl_id=" << order.cl_id << std::endl;
426 
427  // Route based on action and product type
428  if (order.action == "place") {
429  if (order.venue_type == "cex") {
430  if (order.product_type == "spot" || order.product_type == "perpetual") {
431  const auto& details = std::get<CexOrderDetails>(order.details);
432  handle_cex_order(order, details);
433  }
434  } else if (order.venue_type == "dex") {
435  if (order.product_type == "amm_swap") {
436  const auto& details = std::get<AmmSwapDetails>(order.details);
437  handle_amm_swap(order, details);
438  } else if (order.product_type == "clmm_swap") {
439  const auto& details = std::get<ClmmSwapDetails>(order.details);
440  handle_clmm_swap(order, details);
441  }
442  } else if (order.venue_type == "chain") {
443  if (order.product_type == "transfer") {
444  const auto& details = std::get<TransferDetails>(order.details);
445  handle_transfer(order, details);
446  }
447  }
448  } else if (order.action == "cancel") {
449  const auto& details = std::get<CancelDetails>(order.details);
450  handle_cancel(order, details);
451  } else if (order.action == "replace") {
452  const auto& details = std::get<ReplaceDetails>(order.details);
453  handle_replace(order, details);
454  } else {
455  // Reject unknown action
456  ExecutionReport report;
457  report.cl_id = order.cl_id;
458  report.status = "rejected";
459  report.reason_code = "invalid_params";
460  report.reason_text = "Unknown action: " + order.action;
461  report.ts_ns = get_current_time_ns();
462  publish_execution_report(report);
463  }
464 
465  } catch (const std::exception& e) {
466  // Send rejection report
467  ExecutionReport report;
468  report.cl_id = order.cl_id;
469  report.status = "rejected";
470  report.reason_code = "invalid_params";
471  report.reason_text = std::string("Processing error: ") + e.what();
472  report.ts_ns = get_current_time_ns();
473  publish_execution_report(report);
474 
475  std::cerr << "[TradingEngine] Order processing error: " << e.what() << std::endl;
476  }
477 }

920  curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);
921 
922  // Set POST data
923  curl_easy_setopt(curl, CURLOPT_POSTFIELDS, json_payload.c_str());
924  curl_easy_setopt(curl, CURLOPT_POSTFIELDSIZE, json_payload.length());
925 
926  // Perform request
927  CURLcode res = curl_easy_perform(curl);
928 
929  // Cleanup
930  curl_slist_free_all(headers);
931  curl_easy_cleanup(curl);
